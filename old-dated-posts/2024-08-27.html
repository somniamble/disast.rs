here is how I wrote this blog entry:

blog() {
  directory="$HOME/projects/disast.rs/html/src/dated"
  filename="${directory}/$(date +%Y-%m-%d).html"
  TEXT="$@"
  if [ -z "$TEXT" ]; then
    echo "${filename}:"
    cat $filename
  else
    OUTPUT="<p>$TEXT</p>"
    echo "$OUTPUT" >> "$filename"
    echo "wrote '$OUTPUT' to $filename"
  fi

}
